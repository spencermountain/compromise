const fancy_lumping = function(terms) {

    if (tag !== false) {
      let Cl = pos.classMapping[tag] || pos.Term;
      terms[i] = new Cl(a.text + ' ' + b.text, tag);
      terms[i].reason = 'lumped(' + terms[i].reason + ')';
      terms[i - 1] = null;
